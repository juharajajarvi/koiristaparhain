DROP TABLE IF EXISTS Entries;
CREATE TABLE Entries
(
    UserName VARCHAR(30),
    Password VARCHAR(30),
	Created TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
	LastUpdated TIMESTAMP,
	KennelName VARCHAR(20),
    Date VARCHAR(20),
    StarLevel VARCHAR(20),
	Achievements TEXT,
    BankAccount TEXT,
    Dogs TEXT,
    Puppies TEXT,
	PuppiesParents TEXT,
    DeadDogs TEXT,
	Happenings TEXT,
	Hints TEXT
);

DROP TABLE IF EXISTS Feedback;
CREATE TABLE Feedback
(
    Email VARCHAR(60),
	Time TIMESTAMP DEFAULT now(),
	Feedback VARCHAR(5000),
	State VARCHAR(20) DEFAULT 'new'
);
